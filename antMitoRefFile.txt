function Hill_Cooperativity(substrate, Shalve, V, h)
  V*(substrate/Shalve)^h/(1 + (substrate/Shalve)^h);
end

Hill_Cooperativity is "Hill Cooperativity"

function Basal_deacetylation_function(v, SIRT1_activity)

end
